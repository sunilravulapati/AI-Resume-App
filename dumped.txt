import axios from "axios";
import { useNavigate } from "react-router";
import toast from "react-hot-toast";
import useUserStore from "../store/userStore";

export default function Onboarding() {
  const { userRecord, fetchUser } = useUserStore();
  const navigate = useNavigate();

  const selectRole = async (role) => {
    try {
      // Send your MongoDB _id (not clerkId anymore)
      await axios.put("http://localhost:4000/api/user/update-role", {
        userId: userRecord._id, 
        role: role
      });
      
      toast.success(`Welcome! You are now a ${role}`);
      
      // Refresh local store so role is no longer 'pending'
      await fetchUser(); 
      
      if (role === 'student') navigate("/student-dashboard");
      else navigate("/recruiter-dashboard");
    } catch (err) {
      toast.error("Failed to save selection");
    }
  };

  return (
    <div className="flex flex-col items-center justify-center min-h-[80vh]">
      <h1 className="text-3xl font-bold mb-4">Final Step</h1>
      <p className="text-gray-500 mb-8">Tell us how you'll be using the platform.</p>
      <div className="flex gap-8">
        <button 
          onClick={() => selectRole('student')}
          className="p-8 border-2 rounded-xl hover:border-blue-500 hover:bg-blue-50 transition text-center w-64"
        >
          <div className="text-3xl mb-2">🎓</div>
          <h2 className="text-xl font-bold">Student</h2>
          <p className="text-gray-500 text-sm mt-2">Analyze and improve my resume</p>
        </button>
        
        <button 
          onClick={() => selectRole('recruiter')}
          className="p-8 border-2 rounded-xl hover:border-green-500 hover:bg-green-50 transition text-center w-64"
        >
          <div className="text-3xl mb-2">💼</div>
          <h2 className="text-xl font-bold">Recruiter</h2>
          <p className="text-gray-500 text-sm mt-2">Find and score top candidates</p>
        </button>
      </div>
    </div>
  );
}
